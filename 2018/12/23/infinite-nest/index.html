Proprietary Brickwall

As I learned to use Python, I built tools to work with commercial real estate data that I used at my job. Just as I was excitedly preparing to publish and show off my new shiny python tools, I realized that I needed the data to show off those tools but the data was proprietary. The proprietary brickwall stopped me in my tracks.

INFINITE-NEST

"That's ok", I thought. I decided to gather publicly available data from the Securities and Exchange Commission website. From the SEC website I could gather data on Commercial Mortgage-Backed Securities (CMBS). The CMBS data consists of;

  1. Mortgage data that produce the cash flows for the securities.
  2. Commercial property data that make up the collateral for the mortgages.

from bs4 import BeautifulSoup
from bs4 import Tag
from collections import defaultdict
import csv
import re
import os


working_dir = r'C:\Users\Username\Desktop\filename'
os.chdir(working_dir)
filename = 'JPC11C05.html'

with open(filename) as sec:
    soup = BeautifulSoup(sec, 'lxml')

tables = [
    table for table in soup.findAll("table") if 'border' not in table.attrs
]

table_count = len(tables)


# Criteria #1.
# Header is Bold
def is_header(tag, keyword=r'bold'):
    # returns bs4.element.Tag or None
    return tag.find("font", {"style": re.compile(keyword)})


# Criteria #2.
# Does NOT contain "BORDER-BOTTOM" in the style attribute
def unwanted_header(tag):
    for data in tag.findAll('td'):
        if 'colspan' in data.attrs:
            colspan_val = int(data['colspan'])
            if colspan_val > 2:
                return True
            else:
                return False


# Simple function to see if tag object passes Criteria # 1 & 2.
def is_good_header(tag):
    if is_header(tag) and not unwanted_header(tag):
        return True
    else:
        return False


# Criteria #3, Cells must span the same number of columns.
def cells_same_colspan(colspan_dict):
    # reduce colspan dict values
    max_colspan_per_row = [max(val) for val in colspan_dict.values()]
    if max(max_colspan_per_row) == 1:
        return True
    else:
        return False

# -------------------THE FUN PART---------------------------------------------------------------------------------------
# If criteria #3 is not met, we create a nested dictionary from our simple colspan dictionary.
# The nested_colspan_dict is a record of rows and all of their colspan values in each row.
# This object will be used to store all the information needed to make changes to the actual
# cells of data.


def nested_colspan_dict(colspan_dict):
    nested_colspan_obj = {
        key: [colspan_tuple for colspan_tuple in enumerate(value)]
        for key, value in colspan_dict.items()}
    return nested_colspan_obj

# Next, since we may need to extract data from the nested dictionary
# multiple times, it makes sense to create a function.
# This function returns a generator object.
# This is helpful in extracting only the values we need
# and not loading everything into memory if it can be
# avoided.


def colspans_to_fix(nested_colspan):
    for key, value in nested_colspan.items():
        for tupl in value:
            # tuples are (colspan_index, colspan_value)
            if tupl[1] != 1:
                # key = row number
                # tuple = (index_of_cell, colspan_value_for_cell)
                yield {key: tupl}


# A function to write the rows to a file.
def write_rows(current_table, filename, data):
    with open(str(current_table) + filename + '.csv', 'w', newline='') as f:
        writer = csv.writer(f)
        # if row avoids blank rows
        writer.writerows(row for row in data.values() if row)


for i in range(table_count):

    table = tables[i]
    header_rows = defaultdict(dict)
    colspan = {}
    line_count = 0

    for row in table.findAll("tr"):
        colspan[line_count] = [int(td_tag['colspan'])
                               if 'colspan' in td_tag.attrs else 1
                               for td_tag in row.findAll('td')]
        if is_good_header(row) and i % 2 == 0:
            header_rows['headers'][line_count] = [data.font.text for data in row.findAll('td')]
            line_count += 1
        elif is_header(row) is None:
            header_rows['tdata'][line_count] = [data.font.text
                                                if isinstance(data.font, Tag) else ''
                                                for data in row.findAll('td')]
            line_count += 1
        else:
            continue
    if cells_same_colspan(colspan):
        write_rows(current_table=i, filename=' header', data=header_rows['headers'])
        write_rows(current_table=i, filename=' table', data=header_rows['tdata'])
    else:
        nested_cs = nested_colspan_dict(colspan)
        bad_colspans = [cs for cs in colspans_to_fix(nested_cs)]
        headers_only = [h for h in header_rows['headers']]
        while not cells_same_colspan(colspan):
            for colspan_dict in bad_colspans:
                for k, v in colspan_dict.items():
                    row_index = k
                    colspan_ind, colspan_val = v
                    colspan_minus_one = colspan_val - 1
                    index_for_insert = colspan_ind + 1
                    if row_index in headers_only:
                        headers = header_rows['headers'][row_index]
                        for insert in range(colspan_minus_one):
                            headers.insert(index_for_insert, '')
                    else:
                        rows = header_rows['tdata'][row_index]
                        for insert in range(colspan_minus_one):
                            rows.insert(index_for_insert, '')
                colspan[row_index][colspan_ind] -= 1
        else:
            write_rows(current_table=i, filename=' header', data=header_rows['headers'])
            write_rows(current_table=i, filename=' table', data=header_rows['tdata'])
